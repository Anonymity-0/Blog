``` python
class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        if len(s)!=len(t):
            return False
        from collections import defaultdict
        s_dict = defaultdict(int)
        t_dict = defaultdict(int)
        for i in s:
            s_dict[i]+=1
        for i in t:
            t_dict[i]+=1
        return s_dict==t_dict
```

``` python
class Solution:
    def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
        n_set = set(nums1)
        re_set = set()
        for i in nums2:
            if i in n_set:
                re_set.add(i) 
        return list(re_set)
```
